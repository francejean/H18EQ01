CREATE TABLE TYPETRX
(
	CodTypTrx VARCHAR(2) NOT NULL PRIMARY KEY,
	DescTrx VARCHAR(25),
	Nature VARCHAR(2)
)




CREATE TABLE TRX (
	IdTrx int IDENTITY(1,1) NOT NULL PRIMARY KEY,
	DateTrx DATE,
	MontantTrx DECIMAL(10,2),
	Reportee BIT,

	CodTypTrx VARCHAR(2) FOREIGN KEY REFERENCES TYPETRX(CodTypTrx),
	IdReser int FOREIGN KEY REFERENCES RESERVATION(IdReser),
	NoCham VARCHAR(4) FOREIGN KEY REFERENCES CHAMBRE(NoCham),
	IdCli int FOREIGN KEY REFERENCES CLIENT(IdCli)	
)
