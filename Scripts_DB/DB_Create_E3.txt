CREATE TABLE ARRIVE
(
	IdArrive int IDENTITY(1,1) NOT NULL PRIMARY KEY,
	DateArrive DATE,
	
	IdCli int FOREIGN KEY REFERENCES CLIENT(IdCli),
	IdReser int FOREIGN KEY REFERENCES RESERVATION(IdReser),
	NoCham int FOREIGN KEY REFERENCES CHAMBRE(NoCham)
)

CREATE TABLE CLIENT
(
	IdCli int IDENTITY(1,1) NOT NULL PRIMARY KEY,
	Nom VARCHAR(25),
	Adresse VARCHAR(25),
	Telephone VARCHAR(10),
	NoCarte VARCHAR(16),
	TypeCarte VARCHAR(15),
	DatExp DATE,
	SoldeDu DECIMAL(12,2)
)
